function camelize(str) {
    let newArr = [];
    let arr = str.split('-');
    
    arr.forEach(function(item, index) {
    
        if(item.length >= 1) {
            if(index === 0) {
                newArr.push(item);
            } else {
                newArr.push(item[0].toUpperCase() + item.slice(1));
            }
        }
    })
    return newArr.join('');
};

function filterRange(arr, a, b) {
    return arr.filter((item) => {
        if (item < a || item > b) return false;
        return true;
    })
};
